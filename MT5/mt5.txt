  const currentPath = window.location.pathname;

  // Add 'active' to normal nav links
  document.querySelectorAll('.navbar-nav .nav-link').forEach(link => {
    const href = link.getAttribute('href');
    if (!href || href === "#") return;

    const linkPath = new URL(link.href).pathname;
    if (linkPath === currentPath) {
      link.classList.add('active');
    }
  });

  // Add 'active' to dropdown items
  document.querySelectorAll('.dropdown-item').forEach(item => {
    const itemPath = new URL(item.href).pathname;
    if (itemPath === currentPath) {
      item.classList.add('active');

      // Activate dropdown-toggle parent as well
      const dropdownToggle = item.closest('.dropdown')?.querySelector('.dropdown-toggle');
      if (dropdownToggle) dropdownToggle.classList.add('active');
    }
  });

function submitForm() {

  const btn = document.getElementById('ctaBtn');
  btn.classList.add('loading');

  const name   = document.getElementById('name').value.trim();
  const email  = document.getElementById('email').value.trim();
  const answer = document.getElementById('answer').value.trim();

  if (!name || !email || !answer) {
    alert('Please fill all fields');
    btn.classList.remove('loading');
    return;
  }

  fetch('https://signup.mt5simulator.com/record-popup.php', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/x-www-form-urlencoded'
    },
    body: new URLSearchParams({
      name, email, answer
    })
  })
  .then(res => res.json())
  .then(data => {
    if (data.status === 'success') {
      showThankYou();
    } else {
      btn.classList.remove('loading');
      alert('Something went wrong');
    }
  })
  .catch(() => {
    btn.classList.remove('loading');
    alert('Network error');
  });
}

<script>
  const track = document.getElementById('carouselTrack');
  const items = document.querySelectorAll('.carousel-item-custom');
  const dotsContainer = document.getElementById('dots');

  let slideIndex = 1; // Start from first real item
  let slideWidth;
  let visibleCount;

  function updateSlideDimensions() {
    visibleCount = window.innerWidth >= 768 ? 3 : 1;
    slideWidth = items[0].offsetWidth;
    moveToSlide(slideIndex, false);
    createDots();
  }

  function moveToSlide(index, animate = true) {
    if (!animate) track.style.transition = 'none';
    else track.style.transition = 'transform 0.5s ease-in-out';

    const offset = slideWidth * index;
    track.style.transform = `translateX(-${offset}px)`;
    updateDots(index);
  }

  function updateDots(index) {
    const total = items.length - 4; // original items
    const activeDot = (index - 1 + total) % total;

    document.querySelectorAll('.dot').forEach((dot, i) =>
      dot.classList.toggle('active', i === activeDot)
    );
  }

  function createDots() {
    dotsContainer.innerHTML = '';
    const total = items.length - 4;

    for (let i = 0; i < total; i++) {
      const dot = document.createElement('span');
      dot.classList.add('dot');
      if (i === 0) dot.classList.add('active');
      dot.addEventListener('click', () => {
        slideIndex = i + 1;
        moveToSlide(slideIndex);
      });
      dotsContainer.appendChild(dot);
    }
  }

  function nextSlide() {
    slideIndex++;
    moveToSlide(slideIndex);

    // Reset to start (seamless loop)
    if (slideIndex === items.length - 2) {
      setTimeout(() => {
        slideIndex = 1;
        moveToSlide(slideIndex, false);
      }, 500);
    }
  }

  // Initial setup
  window.addEventListener('load', () => {
    updateSlideDimensions();
    moveToSlide(slideIndex, false);
  });

  window.addEventListener('resize', () => {
    updateSlideDimensions();
  });

  // Auto slide
  setInterval(() => {
    nextSlide();
  }, 4000);
</script>

$(document).ready(function() {
  let iti;

  function initValentinePhone() {
    const phoneInput = document.querySelector("#phone");
    if (phoneInput && !phoneInput.classList.contains('iti-added')) {
      iti = window.intlTelInput(phoneInput, {
        initialCountry: "in",
        separateDialCode: true,
        autoPlaceholder: "polite",
        utilsScript: "https://cdnjs.cloudflare.com/ajax/libs/intl-tel-input/17.0.8/js/utils.js",
      });
      phoneInput.classList.add('iti-added');
    }
  }

  const observer = new MutationObserver(function() {
    if (document.querySelector("#phone")) {
      initValentinePhone();
    }
  });

  observer.observe(document.body, { childList: true, subtree: true });
  initValentinePhone();

  $(document).on("submit", "#valentineForm", function(e) {
    e.preventDefault();
    const $form = $(this);
    const $btn = $form.find(".v-submit-btn");
    const originalBtnText = $btn.html();

    let fullPhoneNumber = "";
    let dialCode = "";
    let number = $form.find("#phone").val().trim();

    if (iti) {
      const countryData = iti.getSelectedCountryData();
      dialCode = countryData.dialCode;
    }

    if (dialCode) {
      fullPhoneNumber = "+" + dialCode + number;
    } else {
      fullPhoneNumber = number;
    }

    $btn.prop('disabled', true)
        .html('<span class="v-spinner"></span> Processing...')
        .css('opacity', '0.7');

    const formData = $form.serializeArray();
    formData.push({ name: "full_phone", value: fullPhoneNumber });

    $.ajax({
      url: "https://signup.mt5simulator.com/process_valentine.php",
      type: "POST",
      data: $.param(formData),
      dataType: 'json',
      success: function(response) {
        if(response.status === "success") {
          $("#v-form-container").fadeOut(300, function() {
            $("#v-success-msg").fadeIn();
            $("#v-success-msg .v-submit-btn")
              .text("Close")
              .prop('disabled', false)
              .css('opacity', '1');
          });
        } else {
          alert("Error: " + response.message);
          resetBtn($btn, originalBtnText);
        }
      },
      error: function(xhr) {
        console.error(xhr.responseText);
        alert("Connection error. Please try again.");
        resetBtn($btn, originalBtnText);
      }
    });
  });

  function resetBtn(btn, text) {
    btn.prop('disabled', false).html(text).css('opacity', '1');
  }

  // --- TRIGGER MODAL IMMEDIATELY ---
  openModal(); 
});
